\begin{luacode*}

function fun (t)
local A = 46.23
local beta = 0.2893
local tc = 497.1

local tr = t/tc
  return A*math.sin(beta*tr)*(1-tr)^beta
end

function trigtable ()
for t = 298, 342, 1  do
tex.print(string.format('%1.2f & %1.3f  \\\\',t, fun(t)))
end
end

function draw (tMax,npoints,option)

if option~=[[]] then
  tex.sprint("\\addplot["..option.."] coordinates{")
else
tex.sprint("\\addplot coordinates{")
end
for t = 298, 342, 1  do
  tex.sprint("("..t..","..fun(t)..")")
end

tex.sprint("}")
end

\end{luacode*}
